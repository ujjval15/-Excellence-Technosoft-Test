CREATE DATABASE Userinfo

CREATE TABLE 'Userinfo'.'tblUser'
(
  'U_Id' INT NOT NULL,
  'U_ID' INT NOT NULL AUTO_INCREMENT,
  'U_Name' VARCHAR(45) NOT NULL,
  'U_pass' TINYTEXT NOT NULL,
  PRIMARY KEY (`U_ID`),
)

CREATE TABLE 'Userinfo'.'tblAddresses'
(
 'A_ID' INT NOT NULL,
 'C_ID' INT NOT NULL,
 'A_STREET' VARCHAR(20) NOT NULL,
 'A_PINCODE' INT NOT NULL,
 'A_COUNTRY' VARCHAR(20) NOT NULL,
 'A_STATE' VARCHAR(20) NOT NULL,
 'A_PHONENO INT NOT NULL'
 FOREIGN KEY ('U_Id') REFERENCES Customers(C_Id),
 PRIMARY KEY ('A_ID')
)
